    ```java
    public class ContainerRoutingTerminal extends Container {
        private final RoutingTerminalTile tile;
        private int lastAllocatedChannels;
        private int lastDeviceCount;
        private boolean lastOnline;
        private boolean lastWarning;
        
        public ContainerRoutingTerminal(RoutingTerminalTile tile, EntityPlayer player) {
            this.tile = tile;
        }
        
        @Override
        public void detectAndSendChanges() {
            super.detectAndSendChanges();
            
            if (tile != null) {
                int allocated = tile.getAllocatedChannels();
                int devices = tile.getConnectedDeviceCount();
                boolean online = tile.isOnline();
                boolean warning = tile.isSoftLimitWarning();
                
                // Send updates if changed
                if (allocated != lastAllocatedChannels || 
                    devices != lastDeviceCount ||
                    online != lastOnline ||
                    warning != lastWarning) {
                    
                    for (Object crafter : crafters) {
                        ICrafting icrafting = (ICrafting) crafter;
                        // Use sendProgressBarUpdate with different window IDs for each field
                        icrafting.sendProgressBarUpdate(this, 0, allocated);
                        icrafting.sendProgressBarUpdate(this, 1, devices);
                        icrafting.sendProgressBarUpdate(this, 2, online ? 1 : 0);
                        icrafting.sendProgressBarUpdate(this, 3, warning ? 1 : 0);
                    }
                    
                    lastAllocatedChannels = allocated;
                    lastDeviceCount = devices;
                    lastOnline = online;
                    lastWarning = warning;
                }
            }
        }
        
        @Override
        @SideOnly(Side.CLIENT)
        public void updateProgressBar(int id, int value) {
            switch (id) {
                case 0: allocatedChannels = value; break;
                case 1: connectedDeviceCount = value; break;
                case 2: isOnline = value == 1; break;
                case 3: softLimitWarning = value == 1; break;
            }
        }
        
        @Override
        public boolean canInteractWith(EntityPlayer player) {
            return tile != null && tile.getWorldObj().getTileEntity(tile.xCoord, tile.yCoord, tile.zCoord) == tile
                && player.getDistanceSq(tile.xCoord + 0.5, tile.yCoord + 0.5, tile.zCoord + 0.5) <= 64.0;
        }
        
        // Getters for GUI
        public int getAllocatedChannels() { return allocatedChannels; }
        public int getConnectedDeviceCount() { return connectedDeviceCount; }
        public boolean isOnline() { return isOnline; }
        public boolean isSoftLimitWarning() { return softLimitWarning; }
        public RoutingTerminalTile getTile() { return tile; }
    }
    ```

    ```java
    @SideOnly(Side.CLIENT)
    public class GuiRoutingTerminal extends GuiContainer {
        private final ContainerRoutingTerminal container;
        
        public GuiRoutingTerminal(ContainerRoutingTerminal container) {
            super(container);
            this.container = container;
            this.xSize = 176;
            this.ySize = 120;
        }
        
        @Override
        protected void drawGuiContainerBackgroundLayer(float partialTicks, int mouseX, int mouseY) {
            // Draw background - use gray color similar to AE2
            GL11.glColor4f(1.0F, 1.0F, 1.0F, 1.0F);
            int x = (width - xSize) / 2;
            int y = (height - ySize) / 2;
            
            // Draw simple gray background
            drawRect(x, y, x + xSize, y + ySize, 0xFF333333);
            drawRect(x + 4, y + 4, x + xSize - 4, y + ySize - 4, 0xFF444444);
        }
        
        @Override
        protected void drawGuiContainerForegroundLayer(int mouseX, int mouseY) {
            String title = "Routing Terminal";
            fontRendererObj.drawString(title, 
                (xSize - fontRendererObj.getStringWidth(title)) / 2, 8, 0xFFFFFF);
            
            // Status line
            boolean online = container.isOnline();
            String status = online ? "Status: Online" : "Status: Offline";
            int statusColor = online ? 0x00FF00 : 0xFF0000;
            fontRendererObj.drawString(status, 10, 28, statusColor);
            
            // Statistics
            fontRendererObj.drawString("Allocated Channels: " + container.getAllocatedChannels(), 
                10, 44, 0xFFFFFF);
            fontRendererObj.drawString("Connected Devices: " + container.getConnectedDeviceCount(), 
                10, 56, 0xFFFFFF);
            
            // Warning
            if (container.isSoftLimitWarning()) {
                fontRendererObj.drawString("! High Channel Usage", 10, 72, 0xFFAA00);
            }
        }
    }
    ```
